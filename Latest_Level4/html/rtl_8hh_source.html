58 typedef enum
59 { /* asm: assembly format; r: result; o1: opd1; o2: opd2; op: operator */
60  asm_op, /* Only the operator, no operand */
61  asm_op_o1, /* Only one operand, no result, eg. goto L */
62  asm_op_r, /* Only the result. Operand implicit? */
63  asm_op_r_o1, /* r <- o1 */
64  asm_op_o1_r, /* r <- o1 */
65  asm_op_r_r_o1, /* r <- r op o1 */
66  asm_op_r_o1_o2, /* r <- o1 op o2 */
67  asm_op_o1_o2_r, /* r <- o1 op o2 */
68  asm_op_o1_o2_st, /* for conditional branch*/
69  asm_op_st, /* label instr */
70  asm_nsy, /* not specified yet */
71  asm_op_o1_o2
72 } ASM_Format;

74 typedef enum
75 { /* rtl: intermediate format; r: result; o1: opd1; o2: opd2; op: operator */
76  rtl_op, /* Only the operator, no operand */
77  rtl_op_o1, /* Only one operand, no result, eg. goto L */
78  rtl_r_op, /* Only the result. Operand implicit? */
79  rtl_op_o1_r,
80  rtl_op_r_o1,
81  rtl_r_op_o1, /* r <- o1 */
82  rtl_r_r_op_o1, /* r <- r op o1 */
83  rtl_r_o1_op_o2, /* r <- o1 op o2 */
84  rtl_op_o1_o2_st, /* for conditional branch */
85  rtl_op_st, /* label instr */
86  rtl_nsy, /* not specified yet */
87  rtl_o1_op_o2
